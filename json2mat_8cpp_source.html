9// From json file to rapidjson document object
10rapidjson::Document
11_parseJson(const char* json_path)
12{
13 std::ifstream ifs(json_path);
14 if (!ifs)
15 throw std::string("(_parseJson) Invalid file <") + json_path +
16 std::string(">");
17 rapidjson::IStreamWrapper isw(ifs);
18 rapidjson::Document d;
19 rapidjson::ParseResult res = d.ParseStream(isw);
20 if (!res)
21 throw std::string("(_parseJson) Invalid json <") + json_path +
22 std::string(">");
23 return d;
24}

26// From json file to unordered_map (string->uint)
27std::unordered_map<std::string, unsigned int>
28_uMapJson(const char* json_path,
29 const bool& info,
30 const char* master_key,
31 const char* node_key)
32{
33 unsigned int i(0);
34 auto d = _parseJson(json_path);
35 std::unordered_map<std::string, unsigned int> ht;
36 if (!d.HasMember(master_key))
37 throw std::string("(_uMapJson) Invalid master_key <") + master_key +
38 std::string(">") + std::string(" inside <") + json_path +
39 std::string(">");
40 if (!d[master_key].IsArray())
41 throw std::string("(_uMapJson) No array found at <") + master_key +
42 std::string(">") + std::string(" inside <") + json_path +
43 std::string(">");
44 if (info)
45 fprintf(stdout,
46 "[INFO] Load node(url) inside hash table (string->uint)\n");
47 auto start = std::chrono::high_resolution_clock::now();
48 for (auto& v : d[master_key].GetArray()) {
49 if (!v.IsObject())
50 throw std::string("(_uMapJson) No object found in array at <") +
51 master_key + std::string("><") + node_key +
52 std::string("> inside <") + json_path + std::string(">");
53 if (!v.HasMember(node_key))
54 throw std::string("(_uMapJson) Invalid node_key <") + master_key +
55 std::string("><") + node_key + std::string("> inside <") +
56 json_path + std::string(">");
57 if (!v[node_key].IsString())
58 throw std::string("(_uMapJson) No string found at <") + master_key +
59 std::string("><") + node_key + std::string("> inside <") +
60 json_path + std::string(">");
61 ht[v[node_key].GetString()] = i;
62 ++i;
63 }
64 auto stop = std::chrono::high_resolution_clock::now();
65 auto duration =
66 std::chrono::duration_cast<std::chrono::microseconds>(stop - start);
67 if (info)
68 fprintf(stdout,
69 "[INFO] Time: %lu\u00B5s, Size: %lu, Bucket count: %lu, Load "
70 "factor: %f\n\n",
71 duration.count(),
72 ht.size(),
73 ht.bucket_count(),
74 ht.load_factor());
75 return ht;
76}

78// Create normal adjacency matrix and dangling vector
79std::pair<arma::sp_mat, arma::rowvec>
80adjacencyMatrixJson(const char* json_path,
81 const bool& info,
82 const char* master_key,
83 const char* node_key,
84 const char* nn_key)
85{
86 auto d = _parseJson(json_path);
87 auto ht = _uMapJson(json_path, info, master_key, node_key);
88 arma::sp_mat am(ht.size(), ht.size());
89 arma::rowvec dang(ht.size(), arma::fill::zeros);
90 if (info)
91 fprintf(stdout,
92 "[INFO] Create adjacency matrix and dangling vector from <%s>\n",
93 json_path);
94 auto start = std::chrono::high_resolution_clock::now();
95 for (auto& v : d[master_key].GetArray()) {
96 if (!v.HasMember(nn_key))
97 throw std::string("(adjacencyMatrixJson) Invalid nn_key <") +
98 master_key + std::string("><") + node_key +
99 std::string("> inside <") + json_path + std::string(">");
100 if (!v[nn_key].IsArray())
101 throw std::string("(adjacencyMatrixJson) No array found at <") +
102 master_key + std::string("><") + nn_key +
103 std::string("> inside <") + json_path + std::string(">");
104 std::string node_url = v[node_key].GetString(); // Normalization factor
105 unsigned int nn_size =
106 v[nn_key].GetArray().Size(); // Check dangling nodes
107 if (nn_size == 0)
108 dang(ht[node_url]) = 1.;
109 for (auto& w : v[nn_key].GetArray()) {
110 if (!w.IsString())
111 throw std::string(
112 "(adjacencyMatrixJson) No string found in array at <") +
113 master_key + std::string("><") + nn_key +
114 std::string("> inside <") + json_path + std::string(">");
115 auto nn_url = w.GetString();
116 if (ht.contains(node_url) && ht.contains(nn_url))
117 am(ht[node_url], ht[nn_url]) = (double)1. / nn_size;
118 else
119 throw std::string("(adjacencyMatrixJson) No node <") + nn_url +
120 std::string("> found in hash table generated from <") +
121 json_path + std::string(">, maybe not explored");
122 }
123 }
124 auto stop = std::chrono::high_resolution_clock::now();
125 auto duration =
126 std::chrono::duration_cast<std::chrono::microseconds>(stop - start);
127 if (info)
128 fprintf(stdout,
129 "[INFO] Time: %lu\u00B5s, Size: %llux%llu, Non zero: %llu, "
130 "Dangling: %llu\n\n",
131 duration.count(),
132 am.n_rows,
133 am.n_cols,
134 am.n_nonzero,
135 (long long unsigned int)arma::accu(dang));
136 return { am, dang };
137}
