10PageRank::PageRank(const arma::sp_mat& am,
11 const bool& info,
12 const double& alpha)
13 : _am(am)
14 , _dang(_am.n_cols, arma::fill::zeros)
15 , _rank(_am.n_cols)
16{
17 setAlpha(alpha);
18 if (_am.n_cols != _am.n_rows)
19 throw std::string(
20 "(PageRank::PageRank) Invalid adjacency matrix, not square");
21 if (info)
22 fprintf(stdout,
23 "[INFO] Normalize adjacency matrix, check dangling nodes, init "
24 "rank vector\n");
25 auto start = std::chrono::high_resolution_clock::now();
26 // Normalize adjacency matrix, and populate dangling nodes vector
27 for (long long unsigned int i = 0; i < _am.n_cols; ++i) {
28 double accu = arma::accu(_am.row(i));
29 if (accu != 0)
30 _am.row(i) /= accu;

32 _dang(i) = 1.;
33 }
34 _rank.fill((double)1. / _am.n_cols);
35 auto stop = std::chrono::high_resolution_clock::now();
36 auto duration =
37 std::chrono::duration_cast<std::chrono::microseconds>(stop - start);
38 if (info)
39 fprintf(stdout,
40 "[INFO] Time: %lu\u00B5s, Size: %llu, Dangling nodes: %u\n\n",
41 duration.count(),
42 _rank.n_cols,
43 (unsigned int)arma::accu(_dang));
44}

137// Evolve pagerank of _am with _alpha, and store result in _rank
138void
139PageRank::evoRank(const double& epsilon, const bool& info)
140{
141 if (info)
142 fprintf(stdout, "[INFO] Evolve rank vector using power iteration\n");
143 auto start = std::chrono::high_resolution_clock::now();
144 unsigned int it(0);
145 auto N = _am.n_cols;
146 double wdang;
147 double a = _alpha / N;
148 double b = (1. - _alpha) / N;
149 arma::rowvec rank(N);
150 arma::rowvec ones(N, arma::fill::ones);
151 do {
152 rank = _rank;
153 wdang = arma::dot(rank, _dang);
154 _rank = rank * _am * _alpha + ones * a * wdang + ones * b;
155 ++it;
156 } while (arma::norm(_rank - rank) >= epsilon);
157 auto stop = std::chrono::high_resolution_clock::now();
158 auto duration =
159 std::chrono::duration_cast<std::chrono::microseconds>(stop - start);
160 if (info)
161 fprintf(
162 stdout,
163 "[INFO] Time: %lu\u00B5s, Size: %llu, Dumping factor: %f, Iterations: "
164 "%u\n\n",
165 duration.count(),
166 _am.n_cols,
167 _alpha,
168 it);
169}
